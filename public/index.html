ðŸ”Œ Integration Example

Connect to this MCP server from your LLM applications:

import { Client } from "@modelcontextprotocol/sdk/client/index.js"; import { SSEClientTransport } from "@modelcontextprotocol/sdk/client/sse.js"; const transport = new SSEClientTransport( new URL("${window.location.origin}/sse") ); const client = new Client( { name: "calendar-client", version: "1.0.0" }, { capabilities: {} } ); await client.connect(transport); // Get calendar events for today const events = await client.callTool("getMyCalendarDataByDate", { date: new Date().toISOString().split('T')[0] });
